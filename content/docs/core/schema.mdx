# Invite
<DatabaseTable
  fields={[
    {
      name: "token",
      type: "string",
      description: "Unique token identifying this invite",
      isPrimaryKey: true,
    },
    {
      name: "createdAt",
      type: "Date",
      description: "Date when the invite was created",
    },
    {
      name: "expiresAt",
      type: "Date",
      description: "Date when the invite expires",
    },
    {
      name: "maxUses",
      type: "number",
      description: "Maximum number of times this invite can be used",
    },
    {
      name: "createdByUserId",
      type: "string",
      description: "ID of the user who created the invite",
      isOptional: true,
      isForeignKey: true,
    },
    {
      name: "redirectToAfterUpgrade",
      type: "string",
      description: "URL to redirect after the user upgrades",
    },
    {
      name: "shareInviterName",
      type: "boolean",
      description: "Whether to share the inviter's name with the invitee",
    },
    {
      name: "email",
      type: "string",
      description: "Optional email associated with this invite",
      isOptional: true,
    },
    {
      name: "role",
      type: "string",
      description: "Role assigned when the invite is used",
    },
  ]}
/>

# InviteUse
<DatabaseTable
  fields={[
    {
      name: "inviteId",
      type: "string",
      description: "ID of the invite being used",
      isPrimaryKey: true,
      isForeignKey: true,
    },
    {
      name: "usedAt",
      type: "Date",
      description: "Date when the invite was used",
    },
    {
      name: "usedByUserId",
      type: "string",
      description: "Optional ID of the user who used the invite",
      isOptional: true,
      isForeignKey: true,
    },
  ]}
/>
